##This program for setting the costing run date
REPORT ZSAPRCK58.

DATA: TVARVC.

DATA: GS_VARVC TYPE TVARVC,
      LV_DATE  TYPE DATUM,
      X        TYPE INT8.

PARAMETERS: P_DATE LIKE SY-DATUM.

SELECT SINGLE * INTO GS_VARVC FROM TVARVC WHERE NAME = 'RE_COST'.

IF SY-SUBRC <> 0.
  GS_VARVC-NAME = 'RE_COST'.
  GS_VARVC-TYPE = 'P'.
  GS_VARVC-LOW = '20190101'.
ELSE.
  LV_DATE = GS_VARVC-LOW.
  LV_DATE = LV_DATE + 1.
  IF LV_DATE <= P_DATE.
    GS_VARVC-LOW = LV_DATE.
  ELSE.
    X = 1 / 0.
  ENDIF.
ENDIF.
MODIFY TVARVC FROM GS_VARVC.
WRITE: LV_DATE.
wait up to 2 seconds.
